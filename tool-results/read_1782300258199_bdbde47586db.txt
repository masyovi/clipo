   228тЖТ   228тЖТfunction AnimatedCounter({ value, suffix }: { value: number; suffix: string }) {
   229тЖТ   229тЖТ  const [display, setDisplay] = useState(0);
   230тЖТ   230тЖТ  const ref = useRef<HTMLSpanElement>(null);
   231тЖТ   231тЖТ  const started = useRef(false);
   232тЖТ   232тЖТ
   233тЖТ   233тЖТ  useEffect(() => {
   234тЖТ   234тЖТ    if (started.current) return;
   235тЖТ   235тЖТ    const el = ref.current;
   236тЖТ   236тЖТ    if (!el) return;
   237тЖТ   237тЖТ    const observer = new IntersectionObserver(
   238тЖТ   238тЖТ      ([entry]) => {
   239тЖТ   239тЖТ        if (entry.isIntersecting && !started.current) {
   240тЖТ   240тЖТ          started.current = true;
   241тЖТ   241тЖТ          const duration = 1800;
   242тЖТ   242тЖТ          const steps = 30;
   243тЖТ   243тЖТ          const stepTime = duration / steps;
   244тЖТ   244тЖТ          let current = 0;
   245тЖТ   245тЖТ          const increment = value / steps;
   246тЖТ   246тЖТ          const timer = setInterval(() => {
   247тЖТ   247тЖТ            current += increment;
   248тЖТ   248тЖТ            if (current >= value) {
   249тЖТ   249тЖТ              setDisplay(value);
   250тЖТ   250тЖТ              clearInterval(timer);
   251тЖТ   251тЖТ            } else {
   252тЖТ   252тЖТ              setDisplay(Math.floor(current));
   253тЖТ   253тЖТ            }
   254тЖТ   254тЖТ          }, stepTime);
   255тЖТ   255тЖТ        }
   256тЖТ   256тЖТ      },
   257тЖТ   257тЖТ      { threshold: 0.2 }
   258тЖТ   258тЖТ    );
   259тЖТ   259тЖТ    observer.observe(el);
   260тЖТ   260тЖТ    return () => observer.disconnect();
   261тЖТ   261тЖТ  }, [value]);
   262тЖТ   262тЖТ
   263тЖТ   263тЖТ  const formatNumber = (n: number): string => {
   264тЖТ   264тЖТ    if (n >= 1000000) return `${(n / 1000000).toFixed(1)}M`;
   265тЖТ   265тЖТ    if (n >= 1000) return `${(n / 1000).toFixed(1)}K`;
   266тЖТ   266тЖТ    if (n % 1 !== 0) return n.toFixed(1);
   267тЖТ   267тЖТ    return n.toString();
   268тЖТ   268тЖТ  };
   269тЖТ   269тЖТ
   270тЖТ   270тЖТ  return (
   271тЖТ   271тЖТ    <span ref={ref} className="inline-block">
   272тЖТ   272тЖТ      {formatNumber(display)}{suffix}
   273тЖТ   273тЖТ    </span>
   274тЖТ   274тЖТ  );
   275тЖТ   275тЖТ}
