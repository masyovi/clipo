   232тЖТ   232тЖТfunction AnimatedCounter({ value, suffix }: { value: number; suffix: string }) {
   233тЖТ   233тЖТ  const [display, setDisplay] = useState(0);
   234тЖТ   234тЖТ  const ref = useRef<HTMLSpanElement>(null);
   235тЖТ   235тЖТ  const started = useRef(false);
   236тЖТ   236тЖТ
   237тЖТ   237тЖТ  useEffect(() => {
   238тЖТ   238тЖТ    if (started.current) return;
   239тЖТ   239тЖТ    const el = ref.current;
   240тЖТ   240тЖТ    if (!el) return;
   241тЖТ   241тЖТ    const observer = new IntersectionObserver(
   242тЖТ   242тЖТ      ([entry]) => {
   243тЖТ   243тЖТ        if (entry.isIntersecting && !started.current) {
   244тЖТ   244тЖТ          started.current = true;
   245тЖТ   245тЖТ          const duration = 1800;
   246тЖТ   246тЖТ          const steps = 30;
   247тЖТ   247тЖТ          const stepTime = duration / steps;
   248тЖТ   248тЖТ          let current = 0;
   249тЖТ   249тЖТ          const increment = value / steps;
   250тЖТ   250тЖТ          const timer = setInterval(() => {
   251тЖТ   251тЖТ            current += increment;
   252тЖТ   252тЖТ            if (current >= value) {
   253тЖТ   253тЖТ              setDisplay(value);
   254тЖТ   254тЖТ              clearInterval(timer);
   255тЖТ   255тЖТ            } else {
   256тЖТ   256тЖТ              setDisplay(Math.floor(current));
   257тЖТ   257тЖТ            }
   258тЖТ   258тЖТ          }, stepTime);
   259тЖТ   259тЖТ        }
   260тЖТ   260тЖТ      },
   261тЖТ   261тЖТ      { threshold: 0.2 }
   262тЖТ   262тЖТ    );
   263тЖТ   263тЖТ    observer.observe(el);
   264тЖТ   264тЖТ    return () => observer.disconnect();
   265тЖТ   265тЖТ  }, [value]);
   266тЖТ   266тЖТ
   267тЖТ   267тЖТ  const formatNumber = (n: number): string => {
   268тЖТ   268тЖТ    if (n >= 1000000) return `${(n / 1000000).toFixed(1)}M`;
   269тЖТ   269тЖТ    if (n >= 1000) return `${(n / 1000).toFixed(1)}K`;
   270тЖТ   270тЖТ    if (n % 1 !== 0) return n.toFixed(1);
   271тЖТ   271тЖТ    return n.toString();
   272тЖТ   272тЖТ  };
   273тЖТ   273тЖТ
   274тЖТ   274тЖТ  return (
   275тЖТ   275тЖТ    <span ref={ref} className="inline-block">
   276тЖТ   276тЖТ      {formatNumber(display)}{suffix}
   277тЖТ   277тЖТ    </span>
   278тЖТ   278тЖТ  );
   279тЖТ   279тЖТ}
