    66тЖТ    66тЖТ  if (typeof window === "undefined") return "";
    67тЖТ    67тЖТ  const raw = window.location.pathname.replace(/^\/+|\/+$/g, "");
    68тЖТ    68тЖТ  const valid: AppRoute[] = ["", "login", "register", "dashboard"];
    69тЖТ    69тЖТ  return valid.includes(raw as AppRoute) ? (raw as AppRoute) : "";

    72тЖТ    72тЖТfunction usePathRouter() {
    73тЖТ    73тЖТ  const [path, setPath] = useState<AppRoute>(getInitialPath);
    74тЖТ    74тЖТ
    75тЖТ    75тЖТ  useEffect(() => {
    76тЖТ    76тЖТ    const readPath = () => {
    77тЖТ    77тЖТ      const raw = window.location.pathname.replace(/^\/+|\/+$/g, "");
    78тЖТ    78тЖТ      const valid: AppRoute[] = ["", "login", "register", "dashboard"];
    79тЖТ    79тЖТ      setPath(valid.includes(raw as AppRoute) ? (raw as AppRoute) : "");
    80тЖТ    80тЖТ    };
    81тЖТ    81тЖТ    window.addEventListener("popstate", readPath);
    82тЖТ    82тЖТ    return () => window.removeEventListener("popstate", readPath);
    83тЖТ    83тЖТ  }, []);
    84тЖТ    84тЖТ
    85тЖТ    85тЖТ  const navigate = useCallback((route: AppRoute) => {
    86тЖТ    86тЖТ    const url = route ? `/${route}` : "/";
    87тЖТ    87тЖТ    window.history.pushState({}, "", url);
    88тЖТ    88тЖТ    setPath(route);
    89тЖТ    89тЖТ  }, []);
    90тЖТ    90тЖТ
    91тЖТ    91тЖТ  return { path, navigate };
